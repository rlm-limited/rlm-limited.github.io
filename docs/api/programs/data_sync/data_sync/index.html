    class BaseDataSync:
    """Base class for all Data Sync implementations"""

    fieldNames: list = ["file_path", "seek_position", "last_modified", "file_size"]

    def __init__(self):
        self.config = config()  # type: ignore
        self.data_dir = self.config.data_dir
        self.state_path = self.config.state_path
        self.files_dict: dict[Path, File] = {}
        self.active: bool = True

    def fetch_all_files(self) -> list[Path]:
        """
        Fetch all Files

        returns:
            list[Path]: sorted(datetime wise) list of all CSV file paths in the data directory
        """
        all_files = []
        for dirpath, _, files in os.walk(self.data_dir, topdown=True):
            for file in files:
                if file.endswith(".csv"):
                    all_files.append(Path(os.path.join(dirpath, file)))
        all_files = all_files.sort()
        return all_files

    def extract_data_from_filepath(self, file_path: Path) -> FileInfo | None:
        """
        Helper function to extract device information from the filepath. Will return None
        if unable to exract the relevent data.

        The data directory is intended to be the root, and be arranged in one of two ways:
        1) data/device_type/device_id/data_file.csv
        2) data/site_id/device_type/device_id/data_file.csv

        The latter is when this is run on a server where the files have been synced.
        For the former, the site_id will be taken from the program config instead.

        There are different kinds of files:

        1) The base file is the general data file, which
        has the format: YYYY-MM-DD.csv. This contains data in the standard format:
        [timestamp, variable_name, raw_data, unit, processed_data]

        2) Status files have the format: YYYY-MM-DD.status.csv. This contains data in the format:
        [status_name, status, description, timestamp]
        """

        site_id, device_type, device_id, file_type = None, None, None, None
        info = None

        try:
            parts = list(file_path.parts)

            fn = parts.pop()
            if fnmatch.fnmatch(fn, "*status.csv"):
                file_type = FileTypes.STATUS
            elif fnmatch.fnmatch(fn, "*.csv"):
                file_type = FileTypes.DATA
            else:
                logger.warning(f"Unknown file type: {fn}")
                return None

            device_id = parts.pop()
            device_type = parts.pop()
            if len(parts) > 0:
                site_id = parts.pop()
            else:
                site_id = self.config.site_id

            info = FileInfo(
                site_id=site_id,
                device_type=device_type,
                device_id=device_id,
                file_type=file_type,
            )
        except Exception as e:
            logger.exception(f"Failed to extract data from filepath {file_path}: {e}")

        return info

    def update_state_dict(self, delete_missing=False):
        """Updates the state dictionary with the current file states"""
        all_files = self.fetch_all_files()
        old_files = set(self.files_dict.keys())
        new, unchanged, modified = 0, 0, 0

        for file_path in all_files:
            stat = file_path.stat()
            if file_path not in self.files_dict:
                file = File(
                    file_path=file_path,
                    seek_position=0,
                    last_modified=int(stat.st_mtime),
                    file_size=stat.st_size,
                )
                self.files_dict[file_path] = file
                logger.info(f"New file found: {file=}")
                new += 1
            else:
                file = self.files_dict[file_path]
                old_files.remove(file_path)
                if (
                    file.last_modified == int(stat.st_mtime)
                    and file.file_size == stat.st_size
                ):
                    logger.debug(f"File unchanged: {file=}")
                    unchanged += 1
                    continue
                file.last_modified = int(stat.st_mtime)
                file.file_size = stat.st_size
                logger.info(f"File updated: {file=}")
                modified += 1

        if delete_missing:
            for file_path in old_files:
                del self.files_dict[file_path]

        logger.info(
            f"Updated state dictionary: {len(all_files)} files scanned, {new} new files, {modified} changed files, {unchanged} unchanged files and {len(old_files)} removed files"
        )

    def read_state(self) -> None:
        """Called when creating the FileDriver. Will load the file state from disk"""
        logger.info("Reading File State")
        if not self.state_path.exists():
            logger.info(
                f"State file {self.state_path} does not exist. Creating a new one."
            )
            self.state_path.touch()

        with self.state_path.open() as f:
            reader = csv.DictReader(f)
            for row in reader:
                file = Path(row["file_path"])
                try:
                    file_obj = File(**row)  # type: ignore
                except Exception as e:
                    logger.warning(
                        f"Unable to parse file state for {file}: {row} ({e})"
                    )
                    file_obj = File(file_path=file)
                self.files_dict[file] = file_obj


    def save_state(self) -> None:
        """Called every poll interval. Will save any file specific state to disk"""
        logger.debug("Trying to save file state...")
        with self.state_path.open("w") as f:
            writer = csv.DictWriter(
                f, fieldnames=self.fieldNames, extrasaction="ignore"
            )
            writer.writeheader()
            for file_path, file_obj in self.files_dict.items():
                writer.writerow(file_obj.model_dump(mode="json"))

        logger.debug("Finished saving state")

    def update_file_seek(self, file_path: Path, seek: int) -> None:
        """Updates the seek position for a file"""
        stat = file_path.stat()
        if file_path in self.files_dict:
            file_obj: File = self.files_dict[file_path]
            file_obj.seek_position = seek
            file_obj.last_modified = int(stat.st_mtime)
            file_obj.file_size = stat.st_size
        else:
            self.files_dict[file_path] = File(
                file_path=file_path,
                seek_position=seek,
                last_modified=int(stat.st_mtime),
                file_size=stat.st_size,
            )

    def read_data(self, file: Path, start_seek: int) -> tuple[list[Variable], int]:
        """Reads Data from CSV data file"""

        with file.open() as f:
            f.seek(start_seek)
            reader = csv.DictReader(f, fieldnames=VAR_FILE_FIELD_NAMES)
            data = []
            for row in reader:
                try:
                    if row["unit"] == "None":
                        variable = StringVariable(**row)  # type: ignore
                    elif row["variable_name"] in [
                        "/utility/energy/flow",
                        "/ac/input_mode",
                        "/conf/pv/oc_voltage",
                    ]:
                        continue
                    else:
                        variable = FloatVariable(**row)  # type:ignore
                    # variable = Variable(*row) #type: ignore
                    data.append(variable)
                except Exception as e:
                    logger.error(f"Error parsing row {row}: {e}")
                    continue
            logger.info(f"Finished reading data from {file}")
            # Return the current file position
            current_position = f.tell()
            return data, current_position

    def read_status(self, file: Path, start_seek: int) -> tuple[list[Status], int]:
        """Reads Status from CSV status file"""

        with file.open() as f:
            f.seek(start_seek)
            reader = csv.DictReader(f, fieldnames=STATUS_FILE_FIELD_NAMES)
            data = []
            for row in reader:
                try:
                    status = Status(**row)  # type: ignore

                    continue
            logger.info(f"Finished reading status from {file}")
            # Return the current file position
            current_position = f.tell()
            return data, current_position

    class InfluxDataSync(BaseDataSync):
    """InfluxDB Data Sync Implementation for both Variables and Status"""

    def __init__(self):
        super().__init__()

    def build_variable_payload(self, data: list[Variable], info: FileInfo) -> str:
        """Builds the payload for variable data in InfluxDB line protocol format"""
        payload = list()
        for variable in data:
            line = list()
            try:
                # variable.timestamp is already a datetime object. Leave as seconds
                timestamp_s = int(variable.timestamp.timestamp())
            except Exception as e:
                logger.error(
                    f"Error parsing existing timestamp {variable.timestamp}: {e}"
                )
                # Skip this variable if timestamp can't be parsed
                continue

            match variable:
                case FloatVariable():
                    var = f"float_value={round(variable.processed_data, 2)}"
                case StringVariable():
                    var = f'string_value="{variable.processed_data}"'
                case _:
                    logger.error(f"Unknown variable type: {data}")
                    continue

            line = [
                f"{self.config.influxdb.variable_table_id},",
                f"site_id={info.site_id},",

    async def post_data(self, data: str):
        """Posts data to InfluxDB"""
        if not self.config.influxdb.token:
            logger.error("InfluxDB token is not configured. Please set DATA_SYNC_INFLUXDB__TOKEN environment variable or add it to .env file.")
            raise ValueError("InfluxDB token is required but not configured")
        url = f"{self.config.influxdb.host}?precision={self.config.influxdb.precision}&db={self.config.influxdb.db}"
        headers = {
            "Content-Type": "text/plain",
            "Authorization": f"Bearer {self.config.influxdb.token}",
        }
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    url, data=data, headers=headers, timeout=60
                ) as resp:
                    logger.debug("Posting Data to InfluxDB Server")
                    if resp.status != 204:
                        error_text = await resp.text()
                        logger.error(f"Unable to post data to InfluxDB with status code: {resp.status} and error: {error_text}")
                    logger.info(f"Data posted successfully with status {resp.status}")
                    return resp.status
        except Exception as e:
            logger.error(f"Failed to post data: {e}")
            raise


    async def sync_files(self, post=True):
        """
        Unified sync for both variable and status files.
        Selects the reader and payload builder based on the file_type via a match statement,
        performs shared read/build/post/update_seek logic to avoid duplication.
        """
        logger.info(f"Starting file sync at {datetime.now()}")

        for file, file_obj in self.files_dict.items():
            if file_obj.seek_position == file_obj.file_size:
                logger.debug(f"Skipping {file} as it is fully synced")
                continue

            info = self.extract_data_from_filepath(file)
            if not info:
                logger.info(f"Skipping {file} as it is not a valid format")
                continue

            # hacky fix to stop syncing an old variable format. Marked for removal in next update
            if info.device_type == "epever_hybrid_inverter":
                continue

            # Choose reader and builder based on file type
            match info.file_type:
                case FileTypes.DATA:
                    data, current_seek_position = self.read_data(
                        file=file, start_seek=file_obj.seek_position
                    )
                    payload = self.build_variable_payload(data, info)
                case FileTypes.STATUS:
                    data, current_seek_position = self.read_status(
                        file=file, start_seek=file_obj.seek_position
                    )
                    payload = self.build_status_payload(data, info)
                case _:
                    payload = None
                    logger.debug(f"Unknown file type for {file}, skipping")
                    continue

            if not payload:
                logger.debug(f"No payload built for {file}")
                continue

            # Post if requested
            if post:
                try:
                    response = await self.post_data(payload)
                    if response == 204:
                        logger.info(f"Successfully posted {info.file_type} data from {file}")
                        self.update_file_seek(file, current_seek_position)
                    else:
                        logger.error(f"Failed to post {info.file_type} data from {file}")
                except Exception as e:
                    logger.error(f"DataSyncError while posting {info.file_type} data from {file}: {e}")
                    continue

        logger.info(f"File sync completed at {datetime.now()}")

    async def sync(self, post=True):
        """
        Full sync cycle - syncs both variables and status using unified logic
        """
        logger.info(f"Starting full sync cycle at {datetime.now()}")
        self.read_state()
        self.update_state_dict()

        await self.sync_files(post=post)

        self.save_state()
        logger.info(f"Full sync cycle completed at {datetime.now()}")
