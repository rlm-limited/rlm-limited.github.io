    class DeviceManager:
    """Manages device registration, lifecycle, and execution.

    The DeviceManager maintains a registry of Device instances and handles
    their creation from configuration, registration, execution, and removal.
    It coordinates device operations and manages their async tasks.
    """

    def __init__(self, environment: env.Environment) -> None:
        """Initialize the DeviceManager.

        Args:
            environment: Environment instance providing configuration and context.
        """
        self.environment = environment
        self.devices: dict[str, Device] = {}
        self._tasks: list[asyncio.Task] = []

    def __repr__(self) -> str:
        """Return a string representation of the DeviceManager.

        Returns:
            A string containing the class name and number of managed devices.
        """
        params = ", ".join(
            [
                f"number_devices={len(self.devices)}",
            ]
        )

        return "".join([self.__class__.__name__, "(", params, ")"])

    async def run(self):
        """Initialize and register all devices from configuration.

        Reads the device configuration from the environment and registers
        all configured devices by creating them from their configurations.
        """
        logger.info(f"Entering run for {self}")
        devices_list = self.environment.config.all_devices.devices
        for device_conf in devices_list:
            self.register_device(device_conf.name, device_conf)

        logger.debug(f"Exiting run for {self}")

    def get_device(self, device_name: str) -> Device | None:
        """Retrieve a device from the registry by name.

        Args:
            device_name: The name of the device to retrieve.

        Returns:
            The Device instance if found, None otherwise.
        """
        logger.debug(f"Getting a device with {device_name} id")
        device = self.devices.get(device_name)
        return device

    def register_device(
        self,
        device_name: str,
        device_conf: DeviceConfig | None = None,
        device: Device | None = None,
    ) -> Device | None:
        """Register a device with the manager.

        Retrieves a device if it is already registered, creates it from a
        device configuration if provided, or registers a provided device object.
        If a new device is created or registered, its run task is started.

        Args:
            device_name: The name of the device. Acts as the key in the devices
                dictionary.
            device_conf: Optional device configuration to create the device using
                the device's "from_device_conf" method. Defaults to None.
            device: Optional Device object to register directly, as an alternative
                to creating from device_conf. Defaults to None.

        Returns:
            The Device instance if successful, None if the device cannot be
            created or if neither device_conf nor device is provided.
        """
        dev = self.get_device(device_name)

        if dev:
            logger.debug(f"Device {dev} is already registered")
            return dev

        if not device and device_conf:
            dev = Device.from_device_conf(device_name, self.environment, device_conf)
            if not dev:
                logger.warning(
                    f"Failed to create device {device_name} from device_conf {device_conf}"
                )
                return None
        elif device:
            dev = device
        else:
            logger.error("Neither device nor device_conf provided")
            return None

        self.devices[device_name] = dev
        run_task(dev.run())
        logger.debug(f"{dev} device registered")
        return dev

    async def remove_device(self, device_name: str) -> None:
        """Remove a device from the manager and cancel its tasks.

        Cancels all tasks associated with the device and removes it from
        the device registry.

        Args:
            device_name: The name of the device to remove.
        """
        dev = self.get_device(device_name)
        if dev:
            await dev.cancel()
            del self.devices[device_name]
            logger.info(f"Device {dev} is deleted")
        else:
            logger.info(f"Device {dev} doesn't exist")

    async def cancel(self):
        """Cancel all managed tasks for this DeviceManager.

        Sets the canceled flag and cancels all async tasks associated with
        this manager, then waits for them to complete.
        """
        self.canceled = True
        for task in self._tasks:
            logger.debug(f"Canceling task {task}")
            task.cancel()

        if self._tasks:
            await asyncio.gather(*self._tasks, return_exceptions=True)
        logger.info(f"Cancel finished for {self}")
