    def register_device_api(cls):
    """Register a DeviceApi class to the factory registry.

    This decorator registers a class that implements the DeviceApi protocol
    to the DEVICE_API_FACTORY dictionary, making it available for retrieval
    via `get_device_api()`.

    Args:
        cls: The DeviceApi implementation class to register.

    Returns:
        The class unchanged, allowing it to be used as a decorator.
    """
    logger.warning(f"Registering {cls} to device api factory")
    DEVICE_API_FACTORY[cls.__name__] = cls
    return cls

    def get_device_api(api_name: str) -> "DeviceApi | None":
    """Retrieve a DeviceApi class from the factory registry.

    Args:
        api_name: The name of the DeviceApi class to retrieve.

    Returns:
        The DeviceApi class if found, None otherwise. Logs an error
        if the API name is not found in the registry.
    """
    api = DEVICE_API_FACTORY.get(api_name)
    if not api:
        logger.error(f"api {api_name} not found in {DEVICE_API_FACTORY=}")
    return api

    Variables¶

    devices.api.DEVICE_API_FACTORY: dict[str, DeviceApi] = {} module-attribute ¶

    Factory registry for DeviceApi implementations.

    Maps API class names to their corresponding DeviceApi class implementations. This dictionary is populated by the register_device_api decorator.
