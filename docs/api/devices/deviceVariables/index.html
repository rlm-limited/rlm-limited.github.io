    class DeviceAvailability(BaseModel):
    """Tracks device availability metrics (uptime and downtime).

    Attributes:
        uptime: Total uptime in seconds. Defaults to 0.0.
        downtime: Total downtime in seconds. Defaults to 0.0.
    """

    uptime: float = 0.0
    downtime: float = 0.0

    def serialize(self) -> dict[str, float]:
        """Serialize the device availability to a dictionary.

        Returns:
            Dictionary containing uptime and downtime values.
        """
        return self.model_dump(mode="json")

    def deserialize(self, dev_avail_dict: dict[str, float]):
        """Deserialize device availability from a dictionary.

        Args:
            dev_avail_dict: Dictionary containing uptime and/or downtime values.
        """
        if not dev_avail_dict:
            logger.debug(
                "dev_avail_dict is empty can not deserialize DeviceAvailability"
            )
            return
        for k, v in dev_avail_dict.items():
            setattr(self, k, v)

    class Vars(BaseModel):
    """Collection of device variables organized by variable name.

    Attributes:
        vars_dict: Dictionary mapping variable names to DeviceVariable instances.
    """

    vars_dict: dict[str, DeviceVariable] = {}

    def serialize(self):
        """Serialize all variables in the collection to a dictionary.

        Returns:
            Dictionary mapping variable names to their serialized representations.
        """
        json_ready = {}
        for k, v in self.vars_dict.items():
            json_ready[k] = v.serialize()
        return json_ready

    def deserialize(self, json_dict: dict[str, dict], device_variables_confs: dict):
        """Deserialize variables from a dictionary representation.

        Restores DeviceVariable instances from serialized data, using the
        provided device variable configurations.

        Args:
            json_dict: Dictionary mapping variable names to their serialized data.
            device_variables_confs: Dictionary mapping variable names to their
                VariablesConfs configurations.
        """
        if not (json_dict or device_variables_confs):
            logger.info("please provide both vars and device_variables_confs")
            return
        for k, v in json_dict.items():
            if k in device_variables_confs:
                try:
                    self.vars_dict[k] = DeviceVariable(**v)
                except Exception as e:
                    logger.warning(f"can not restore ({k}), {e}")
                    continue
