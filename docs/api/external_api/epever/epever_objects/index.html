    class EpeverVariableObject(DeviceVariable):
    """Base class for Epever variable objects that interpret Modbus register data.

    This class provides the core functionality for processing raw Modbus register
    values from Epever inverters. Subclasses must implement `interpret_register()`
    to define how raw register data is converted to meaningful values.

    Attributes:
        _variable_confs: EpeverVariablesConfs instance containing configuration
            for this variable.
    """

    def interpret_register(self) -> str | int | float:
        """Interpret raw register data into a processed value.

        This method must be implemented by subclasses to define how raw register
        data is converted to the appropriate data type (string, int, or float).

        Returns:
            The interpreted value as a string, integer, or float.

        Raises:
            NotImplementedError: If not implemented by subclass.
        """
        ...

    def extract_bits(self, value, start_bit, end_bit=None):
        """Extract bits from a value within a specified bit range.

        Args:
            value: The integer value to extract bits from.
            start_bit: Starting bit position (0-indexed from LSB).
            end_bit: Ending bit position (inclusive). If None, extracts a
                single bit at start_bit. Defaults to None.

        Returns:
            The extracted bit value as an integer. If end_bit is None,
            returns 0 or 1. Otherwise, returns the value of the bit range.
        """
        if end_bit is None:
            return (value >> start_bit) & 1
        mask = (1 << (end_bit - start_bit + 1)) - 1
        return (value >> start_bit) & mask

    def coeff_value(self, flt=False) -> int | float:
        """Apply coefficient scaling to raw data.

        Divides the raw register data by the coefficient to get the actual
        physical value. The coefficient is used to scale integer register
        values to their real-world units.

        Args:
            flt: If True, returns a float rounded to 2 decimal places.
                If False, returns an integer. Defaults to False.

        Returns:
            The scaled value as an integer or float, depending on the flt
            parameter.
        """
        if flt:
            return round(self.raw_data / self._variable_confs.coefficient, 2)
        return int(self.raw_data / self._variable_confs.coefficient)

    @classmethod
    def from_register(
        cls,
        variable_name: str,
        raw_data: int,
        timestamp: datetime,
        variable_confs: EpeverVariablesConfs,
    ) -> Self:
        """Create an EpeverVariableObject instance from Modbus register data.

        Factory method that creates and initializes a variable object from raw
        register data, automatically processing the data according to the
        variable configuration.

        Args:
            variable_name: Name identifier for this variable.
            raw_data: Raw integer value read from the Modbus register.
            timestamp: Timestamp when the data was read.
            variable_confs: EpeverVariablesConfs instance containing
                configuration for interpreting the register data.

        Returns:
            An initialized EpeverVariableObject instance with processed data.
        """
        obj = cls(
            variable_name=variable_name,
            raw_data=raw_data,
            unit=variable_confs.unit,
            timestamp=timestamp,
        )
        obj._variable_confs = variable_confs
        obj.process_data()
        return obj

    def pretty_print(self) -> str:
        """Generate a human-readable string representation of the variable.

        Formats the raw data value with its unit of measurement if available.

        Returns:
            A formatted string containing the raw data value and unit,
            or just the raw data value if no unit is configured.
        """
        if self._variable_confs.unit:
            return f"{self.raw_data} {self._variable_confs.unit}"
        return self.raw_data

    def process_data(self) -> None:
        """Process raw register data into interpreted value.

        Calls `interpret_register()` to convert raw data to processed data
        and stores the result in `processed_data`.
        """
        self.processed_data = self.interpret_register()
        return None

    @register_variable_objects
class StartAndEndBit(EpeverVariableObject):
    """Variable object for extracting a bit range and mapping to string values.

    Extracts a range of bits from the register value (using start_bit and
    end_bit) and maps the resulting integer to a string representation
    using the configured mapping dictionary.
    """

    def interpret_register(self) -> str:
        """Interpret register by extracting bit range and mapping to string.

        Returns:
            The mapped string value from the configuration, or "Unknown (value)"
            if no mapping is configured or the value is not in the mapping.
        """
        value = self.extract_bits(
            self.coeff_value(),
            self._variable_confs.start_bit,
            self._variable_confs.end_bit,
        )
        if not self._variable_confs.mapping:
            logger.warning(f"Missing mapping conf for {self.variable_name}")
            return f"Unknown ({value})"
        return self._variable_confs.mapping.get(value, f"Unknown ({value})")

    @register_variable_objects
class StandardFloatVariableSigned(EpeverVariableObject):
    """Variable object for signed float values.

    Interprets register data as a signed floating-point value using two's
    complement representation. Values with the MSB (bit 15) set are treated
    as negative numbers.
    """

    def interpret_register(self) -> float:
        """Interpret register as signed float value.

        Checks if the most significant bit (bit 15) is set to determine
        if the value is negative, then applies two's complement conversion
        and coefficient scaling.

        Returns:
            The scaled signed float value rounded to 2 decimal places.
        """
        if self.raw_data & 32768 == 32768:
            signed_number = self.raw_data - 65536
            return signed_number / self._variable_confs.coefficient
        return self.coeff_value(flt=True)

    def get_variable_cls(variable_name: str) -> type["EpeverVariableObject"] | None:
    """Retrieve a variable object class from the factory registry.

    Args:
        variable_name: The name of the variable class to retrieve.

    Returns:
        The registered EpeverVariableObject subclass if found, None otherwise.

    cls = VARIABLE_FACTORY.get(variable_name, None)

    return cls
